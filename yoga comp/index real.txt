let current = 0;
let score = 0;
let time = 30;
let timer;

loadQuestion();

function loadQuestion(){

    clearInterval(timer);

    document.getElementById("nextBtn").style.display="none";

    time = 30;

    document.getElementById("timer").innerHTML =
    "⏳ Time Left : 30 sec";

    document.getElementById("qno").innerHTML =
    "Question " + (current+1) + " of " + quiz.length;

    document.getElementById("question").innerHTML =
    quiz[current].question;

    let images =
    document.getElementsByClassName("optionImg");

    let currentOptions = [...quiz[current].options];

// Randomly shuffle the two images
currentOptions.sort(() => Math.random() - 0.5);

for(let i=0;i<2;i++){

    images[i].src =
    currentOptions[i];

    images[i].className =
    "optionImg";

    images[i].style.pointerEvents="auto";

    images[i].onclick=function(){
        checkAnswer(this);
    };
}
    startTimer();
}

function startTimer(){

    timer=setInterval(function(){

        time--;

        document.getElementById("timer").innerHTML =
        "⏳ Time Left : " + time + " sec";

        if(time<=0){

            clearInterval(timer);

            showCorrect();

            document.getElementById("nextBtn")
            .style.display="inline-block";
        }

    },1000);
}

function showCorrect(){

    let correct = quiz[current].answer;

    let images =
    document.getElementsByClassName("optionImg");

    for(let img of images){

        img.style.pointerEvents="none";

        if(img.src.includes(correct.split('/').pop())){

            img.classList.add("correct");
        }
    }
}

function checkAnswer(selected){

    clearInterval(timer);

    let correct = quiz[current].answer;

    let images =
    document.getElementsByClassName("optionImg");

    for(let img of images){

        img.style.pointerEvents="none";

        if(img.src.includes(correct.split('/').pop())){

            img.classList.add("correct");
        }
    }

    if(selected.src.includes(correct.split('/').pop())){

        score++;

        document.getElementById("score").innerHTML =
        "Score : " + score;
    }
    else{

        selected.classList.add("wrong");
    }

    document.getElementById("nextBtn")
    .style.display="inline-block";
}

function nextQuestion(){

    current++;

    if(current < quiz.length){

        loadQuestion();
    }
    else{

        clearInterval(timer);

        document.getElementById("qno").style.display="none";
        document.getElementById("question").style.display="none";
        document.getElementById("options").style.display="none";
        document.getElementById("timer").style.display="none";
        document.getElementById("nextBtn").style.display="none";

        document.getElementById("restartBtn")
        .style.display="inline-block";

        document.getElementById("result").innerHTML =
        "🎉 QUIZ COMPLETED 🎉<br><br>Your Score : "
        + score + " / " + quiz.length;
    }
}

function restartQuiz(){

    current = 0;
    score = 0;

    document.getElementById("score").innerHTML =
    "Score : 0";

    document.getElementById("qno").style.display="block";
    document.getElementById("question").style.display="block";
    document.getElementById("options").style.display="flex";
    document.getElementById("timer").style.display="block";

    document.getElementById("restartBtn")
    .style.display="none";

    document.getElementById("result").innerHTML="";

    loadQuestion();
}
